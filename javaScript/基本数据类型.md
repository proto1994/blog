4. 不可以使用yield命令，箭头函数不能当作generator函数



柯里化函数：

```js
function curry(fn) {
  return function dfs(...args) {
    if (args.length > fn.length) {
      return fn.apply(null, args);
    }
    return (...arg1) => {
      return dfs.apply(null, args.concat(arg1))
    }
  }
}

const sum = curry(function(a, b, c) {
  return a + b + c;
})

sum(0)(1)(2)
```
